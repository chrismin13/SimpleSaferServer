#!/bin/bash # Clone the GitHub repository to a temporary directory git clone https://github.com/chrismin13/SimpleSaferServer /tmp/SimpleSaferServer # Navigate to the directory containing the install script cd /tmp/SimpleSaferServer # Check if script is running as root and if not, then prompt for elevation if [ "$EUID" -ne 0 ]; then echo "This script must be run as root. Attempting to elevate privilege..." sudo bash install.sh else bash install.sh fi # Navigate back to the original directory and remove the temporary one cd - rm -rf /tmp/SimpleSaferServer
